import math
from typing import Any, Dict, Optional, Tuple, List

import torch
import torch.nn as nn
import torch.nn.functional as F

# ==========================
# Masked Multi-Head Self-Attention (MHSA) Module
# ==========================
class MaskedMHSA(nn.Module):
    """Masked Multi-Head Self-Attention module that attends only over valid tokens.

    Attributes:
        hidden_dim (int): Dimensionality of token embeddings.
        num_heads (int): Number of attention heads.
        dropout (float): Dropout probability.
    """
    def __init__(self, hidden_dim: int, num_heads: int = 8, dropout: float = 0.1) -> None:
        super(MaskedMHSA, self).__init__()
        self.hidden_dim: int = hidden_dim
        self.num_heads: int = num_heads
        if hidden_dim % num_heads != 0:
            raise ValueError("hidden_dim must be divisible by num_heads")
        self.head_dim: int = hidden_dim // num_heads
        self.scale: float = math.sqrt(self.head_dim)
        self.qkv: nn.Linear = nn.Linear(hidden_dim, hidden_dim * 3)
        self.out_proj: nn.Linear = nn.Linear(hidden_dim, hidden_dim)
        self.dropout: nn.Dropout = nn.Dropout(dropout)

    def forward(self, x: torch.Tensor, attn_mask: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x (torch.Tensor): Input tensor of shape [B, L, hidden_dim].
            attn_mask (torch.Tensor): Attention mask of shape [B, L],
                where valid tokens are 1.0 and padded tokens are 0.0.
        Returns:
            torch.Tensor: Output tensor of shape [B, L, hidden_dim].
        """
        B, L, _ = x.shape
        qkv = self.qkv(x)  # [B, L, 3 * hidden_dim]
        qkv = qkv.reshape(B, L, 3, self.num_heads, self.head_dim)
        q, k, v = qkv.unbind(dim=2)  # each is [B, L, num_heads, head_dim]
        q = q.transpose(1, 2)  # [B, num_heads, L, head_dim]
        k = k.transpose(1, 2)
        v = v.transpose(1, 2)
        attn_scores = torch.matmul(q, k.transpose(-2, -1)) / self.scale  # [B, num_heads, L, L]
        # Create mask: valid tokens (1.0) remain, padded tokens (0.0) become -infty.
        mask = (1.0 - attn_mask).unsqueeze(1).unsqueeze(2)  # [B, 1, 1, L]
        attn_scores = attn_scores + mask * (-1e9)
        attn_probs = F.softmax(attn_scores, dim=-1)
        attn_probs = self.dropout(attn_probs)
        attn_output = torch.matmul(attn_probs, v)  # [B, num_heads, L, head_dim]
        attn_output = attn_output.transpose(1, 2).reshape(B, L, self.hidden_dim)
        output = self.out_proj(attn_output)
        return output

# ==========================
# SwiGLU Feed-Forward Network (FFN) Module
# ==========================
class SwiGLU(nn.Module):
    """SwiGLU Feed-Forward Network block replacing the standard MLP.
    
    Implements the function:
       f(x) = Linear( SiLU(W₁(x)) ⊙ W₂(x) )
    where the two projections (W₁ and W₂) are computed together and then split.
    """
    def __init__(self, hidden_dim: int, dropout: float = 0.1) -> None:
        super(SwiGLU, self).__init__()
        # Project to 2*hidden_dim, then split into two parts.
        self.fc1: nn.Linear = nn.Linear(hidden_dim, hidden_dim * 2)
        self.fc2: nn.Linear = nn.Linear(hidden_dim, hidden_dim)
        self.dropout: nn.Dropout = nn.Dropout(dropout)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x (torch.Tensor): Input tensor of shape [B, L, hidden_dim].
        Returns:
            torch.Tensor: Output tensor of shape [B, L, hidden_dim].
        """
        x_proj = self.fc1(x)  # [B, L, 2*hidden_dim]
        a, b = x_proj.chunk(2, dim=-1)  # each [B, L, hidden_dim]
        activated = F.silu(a) * b
        activated = self.dropout(activated)
        output = self.fc2(activated)
        return output

# ==========================
# Transformer Block Module for FiT
# ==========================
class TransformerBlock(nn.Module):
    """Transformer block with LayerNorm, Masked MHSA, and SwiGLU FFN with residual connections."""
    def __init__(self, hidden_dim: int, num_heads: int = 8, dropout: float = 0.1) -> None:
        super(TransformerBlock, self).__init__()
        self.norm1: nn.LayerNorm = nn.LayerNorm(hidden_dim)
        self.attn: MaskedMHSA = MaskedMHSA(hidden_dim, num_heads, dropout)
        self.norm2: nn.LayerNorm = nn.LayerNorm(hidden_dim)
        self.ffn: SwiGLU = SwiGLU(hidden_dim, dropout)

    def forward(self, x: torch.Tensor, attn_mask: torch.Tensor) -> torch.Tensor:
        """
        Args:
            x (torch.Tensor): Input tensor of shape [B, L, hidden_dim].
            attn_mask (torch.Tensor): Attention mask of shape [B, L].
        Returns:
            torch.Tensor: Output tensor of shape [B, L, hidden_dim].
        """
        x = x + self.attn(self.norm1(x), attn_mask)
        x = x + self.ffn(self.norm2(x))
        return x

# ==========================
# FiT Model: Flexible Vision Transformer for Diffusion
# ==========================
class Model(nn.Module):
    """
    Flexible Vision Transformer (FiT) for Diffusion Models.
    
    This model integrates a pretrained VAE for encoding images,
    patchifies the latent output into a sequence of tokens using a configurable
    strategy (default: center crop), applies a linear projection to the tokens,
    adds 2D RoPE positional embeddings (with support for resolution extrapolation),
    optionally incorporates extra conditioning signals via projection, and processes
    tokens through stacked transformer blocks using Masked MHSA and SwiGLU.
    
    Public Methods:
      - __init__(params: Dict[str, Any])
      - forward(x: torch.Tensor, cond: Optional[torch.Tensor] = None, target_resolution: Optional[Tuple[int, int]] = None) -> Tuple[torch.Tensor, torch.Tensor]
      - load_pretrained_vae(vae_path: str) -> None
    """
    def __init__(self, params: Dict[str, Any]) -> None:
        """
        Args:
            params (Dict[str, Any]): Configuration parameters (from config.yaml).
        """
        super(Model, self).__init__()
        # Load model configuration parameters.
        model_config: Dict[str, Any] = params.get("model", {})
        self.patch_size: int = int(model_config.get("patch_size", 2))
        self.max_token_length: int = int(model_config.get("max_token_length", 256))
        # Patching strategy: "crop" (default) or "pool"
        self.patching_strategy: str = model_config.get("patching_strategy", "crop").lower()
        
        # Transformer configuration.
        transformer_config: Dict[str, Any] = model_config.get("transformer", {})
        self.hidden_dim: int = int(transformer_config.get("hidden_dim", 512))
        self.num_layers: int = int(transformer_config.get("num_layers", 12))
        self.num_heads: int = int(transformer_config.get("num_heads", 8))
        self.dropout: float = float(transformer_config.get("dropout", 0.1))
        
        # Positional embedding interpolation method.
        extrapolation_config: Dict[str, Any] = params.get("extrapolation", {})
        # Default to "VisionNTK" if not specified.
        self.positional_interpolation_method: str = extrapolation_config.get("default_method", "VisionNTK")
        
        # Load the pretrained VAE.
        pretrained_vae_path: str = model_config.get("pretrained_vae", "huggingface/stabilityai/sd-vae-ft-ema")
        self.vae: nn.Module = self.load_pretrained_vae(pretrained_vae_path)
        self.vae.eval()  # Use evaluation mode for VAE.
        
        # Assume latent channels from VAE; for DummyVAE, channels = 4.
        self.latent_channels: int = 4
        # Compute token dimension: latent_channels * (patch_size^2).
        self.token_dim: int = self.latent_channels * (self.patch_size ** 2)
        
        # Linear projection to map patch tokens to transformer hidden dimension.
        self.token_proj: nn.Linear = nn.Linear(self.token_dim, self.hidden_dim)
        
        # Optional conditioning projection (e.g., for diffusion timesteps).
        cond_dim: Optional[Any] = model_config.get("cond_dim", None)
        if cond_dim is not None:
            self.cond_proj: nn.Linear = nn.Linear(int(cond_dim), self.hidden_dim)
        else:
            self.cond_proj = None
        
        # Build the stacked transformer blocks.
        self.transformer_blocks: nn.ModuleList = nn.ModuleList([
            TransformerBlock(self.hidden_dim, self.num_heads, self.dropout)
            for _ in range(self.num_layers)
        ])
        
        # Final layer normalization.
        self.final_norm: nn.LayerNorm = nn.LayerNorm(self.hidden_dim)
    
    def forward(
        self,
        x: torch.Tensor,
        cond: Optional[torch.Tensor] = None,
        target_resolution: Optional[Tuple[int, int]] = None
    ) -> Tuple[torch.Tensor, torch.Tensor]:
        """
        Args:
            x (torch.Tensor): Input image tensor of shape [B, C, H, W].
            cond (Optional[torch.Tensor]): Optional extra conditioning signals of shape [B, cond_dim].
            target_resolution (Optional[Tuple[int, int]]): Target resolution (H_test, W_test) for positional extrapolation.
        Returns:
            Tuple[torch.Tensor, torch.Tensor]: 
                - Transformer output tokens [B, max_token_length, hidden_dim].
                - Attention mask [B, max_token_length] with 1.0 for valid tokens.
        """
        # 1. Encode image to latent space using the pretrained VAE.
        latent: torch.Tensor = self._encode(x)  # [B, latent_channels, H_latent, W_latent]
        # 2. Patchify latent representation into tokens, obtain token mask and spatial coordinates.
        tokens, attn_mask, coords = self._patchify(latent)  # tokens: [B, L, token_dim]
        # 3. Project patch tokens to the transformer hidden space.
        token_embeddings: torch.Tensor = self.token_proj(tokens)  # [B, L, hidden_dim]
        # 4. Compute 2D RoPE positional embeddings from token coordinates.
        pos_embeddings: torch.Tensor = self._get_2d_rope_positional_embedding(coords, target_resolution)  # [B, L, hidden_dim]
        # 5. Combine token embeddings with positional embeddings.
        x_embed: torch.Tensor = token_embeddings + pos_embeddings
        # 6. Incorporate extra conditioning signals if provided.
        if cond is not None and self.cond_proj is not None:
            cond_embedding: torch.Tensor = self.cond_proj(cond)  # [B, hidden_dim]
            x_embed = x_embed + cond_embedding.unsqueeze(1)  # Broadcast to [B, L, hidden_dim]
        # 7. Process combined embeddings through transformer blocks.
        for block in self.transformer_blocks:
            x_embed = block(x_embed, attn_mask)
        x_embed = self.final_norm(x_embed)
        return x_embed, attn_mask

    def load_pretrained_vae(self, vae_path: str) -> nn.Module:
        """
        Public method to load a pretrained VAE model.
        
        Args:
            vae_path (str): Path or identifier for the pretrained VAE.
        Returns:
            nn.Module: Pretrained VAE model.
        """
        # DummyVAE simulates a pretrained VAE.
        class DummyVAE(nn.Module):
            def __init__(self) -> None:
                super(DummyVAE, self).__init__()
                self.encoder: nn.Sequential = nn.Sequential(
                    nn.Conv2d(3, 4, kernel_size=3, stride=2, padding=1),
                    nn.ReLU()
                )
                self.decoder: nn.Sequential = nn.Sequential(
                    nn.ConvTranspose2d(4, 3, kernel_size=3, stride=2, padding=1, output_padding=1),
                    nn.Sigmoid()
                )
            def encode(self, x: torch.Tensor) -> torch.Tensor:
                return self.encoder(x)
            def decode(self, z: torch.Tensor) -> torch.Tensor:
                return self.decoder(z)
        return DummyVAE()

    # --------------------------
    # Private Helper Methods
    # --------------------------
    def _encode(self, x: torch.Tensor) -> torch.Tensor:
        """
        Private method to encode the input image using the pretrained VAE.
        
        Args:
            x (torch.Tensor): Input image tensor [B, C, H, W].
        Returns:
            torch.Tensor: Latent representation [B, latent_channels, H_latent, W_latent].
        """
        latent = self.vae.encode(x)
        if isinstance(latent, (tuple, list)):
            latent = latent[0]
        return latent

    def _decode(self, latent: torch.Tensor) -> torch.Tensor:
        """
        Private method to decode latent representation using the pretrained VAE.
        
        Args:
            latent (torch.Tensor): Latent representation.
        Returns:
            torch.Tensor: Reconstructed image.
        """
        return self.vae.decode(latent)

    def _patchify(self, latent: torch.Tensor) -> Tuple[torch.Tensor, torch.Tensor, torch.Tensor]:
        """
        Private method to convert latent maps into a sequence of patch tokens.
        
        Steps:
          - Pad latent maps so height and width are divisible by patch_size.
          - Extract non-overlapping patches via nn.Unfold.
          - Reshape patches to grid form and apply the patching strategy:
              • If total tokens > max_token_length, apply center cropping (default)
                or pooling (if configured) to reduce tokens.
              • Otherwise, pad tokens to reach max_token_length.
          - Generate an attention mask (1.0 for real tokens, 0.0 for paddings).
          - Compute corresponding (h, w) grid coordinates.
        
        Args:
            latent (torch.Tensor): Latent tensor [B, C, H, W].
        Returns:
            Tuple[torch.Tensor, torch.Tensor, torch.Tensor]:
                tokens: [B, max_token_length, token_dim]
                attn_mask: [B, max_token_length] (1.0 for valid tokens)
                coords: [B, max_token_length, 2] grid coordinates.
        """
        B, C, H, W = latent.shape
        pad_h = (self.patch_size - (H % self.patch_size)) % self.patch_size
        pad_w = (self.patch_size - (W % self.patch_size)) % self.patch_size
        if pad_h > 0 or pad_w > 0:
            latent = F.pad(latent, (0, pad_w, 0, pad_h), mode="constant", value=0)
            H += pad_h
            W += pad_w
        unfold = nn.Unfold(kernel_size=self.patch_size, stride=self.patch_size)
        patches = unfold(latent)  # [B, token_dim, L_total]
        L_total = patches.shape[-1]
        H_grid = H // self.patch_size
        W_grid = W // self.patch_size
        patches_grid = patches.view(B, self.token_dim, H_grid, W_grid).permute(0, 2, 3, 1).contiguous()
        # Create coordinate grid: [H_grid, W_grid, 2]
        coord_y = torch.arange(H_grid, device=latent.device).unsqueeze(1).expand(H_grid, W_grid)
        coord_x = torch.arange(W_grid, device=latent.device).unsqueeze(0).expand(H_grid, W_grid)
        grid_coords = torch.stack((coord_y, coord_x), dim=-1)
        tokens_list: List[torch.Tensor] = []
        mask_list: List[torch.Tensor] = []
        coords_list: List[torch.Tensor] = []
        for b in range(B):
            tokens_sample = patches_grid[b]  # [H_grid, W_grid, token_dim]
            coords_sample = grid_coords  # [H_grid, W_grid, 2]
            num_tokens = tokens_sample.numel() // self.token_dim
            if num_tokens > self.max_token_length:
                if self.patching_strategy == "crop":
                    crop_h = min(H_grid, int(math.sqrt(self.max_token_length)))
                    crop_w = min(W_grid, self.max_token_length // crop_h)
                    start_h = (H_grid - crop_h) // 2
                    start_w = (W_grid - crop_w) // 2
                    tokens_cropped = tokens_sample[start_h:start_h + crop_h, start_w:start_w + crop_w]
                    coords_cropped = coords_sample[start_h:start_h + crop_h, start_w:start_w + crop_w]
                    tokens_flat = tokens_cropped.reshape(-1, self.token_dim)
                    coords_flat = coords_cropped.reshape(-1, 2)
                elif self.patching_strategy == "pool":
                    # Downsample tokens via adaptive average pooling.
                    tokens_temp = tokens_sample.permute(2, 0, 1).unsqueeze(0)
                    tokens_pooled = F.adaptive_avg_pool2d(tokens_temp, (1, self.max_token_length))
                    tokens_flat = tokens_pooled.squeeze(0).permute(1, 2, 0).reshape(-1, self.token_dim)
                    # For coordinates, fallback to cropping.
                    crop_h = min(H_grid, int(math.sqrt(self.max_token_length)))
                    crop_w = min(W_grid, self.max_token_length // crop_h)
                    start_h = (H_grid - crop_h) // 2
                    start_w = (W_grid - crop_w) // 2
                    coords_cropped = coords_sample[start_h:start_h + crop_h, start_w:start_w + crop_w]
                    coords_flat = coords_cropped.reshape(-1, 2)
                else:
                    raise ValueError(f"Unknown patching strategy: {self.patching_strategy}")
            else:
                tokens_flat = tokens_sample.reshape(-1, self.token_dim)
                coords_flat = coords_sample.reshape(-1, 2)
            current_len = tokens_flat.shape[0]
            if current_len < self.max_token_length:
                pad_len = self.max_token_length - current_len
                pad_tokens = torch.zeros((pad_len, self.token_dim), device=latent.device, dtype=tokens_flat.dtype)
                tokens_flat = torch.cat([tokens_flat, pad_tokens], dim=0)
                pad_coords = torch.zeros((pad_len, 2), device=latent.device, dtype=coords_flat.dtype)
                coords_flat = torch.cat([coords_flat, pad_coords], dim=0)
                attn_mask = torch.cat([torch.ones(current_len, device=latent.device), torch.zeros(pad_len, device=latent.device)], dim=0)
            else:
                tokens_flat = tokens_flat[:self.max_token_length]
                coords_flat = coords_flat[:self.max_token_length]
                attn_mask = torch.ones(self.max_token_length, device=latent.device)
            tokens_list.append(tokens_flat)
            mask_list.append(attn_mask)
            coords_list.append(coords_flat)
        tokens_final = torch.stack(tokens_list, dim=0)         # [B, max_token_length, token_dim]
        mask_final = torch.stack(mask_list, dim=0)               # [B, max_token_length]
        coords_final = torch.stack(coords_list, dim=0)           # [B, max_token_length, 2]
        return tokens_final, mask_final, coords_final

    def _get_2d_rope_positional_embedding(
        self, 
        coords: torch.Tensor, 
        target_resolution: Optional[Tuple[int, int]] = None
    ) -> torch.Tensor:
        """
        Computes 2D Rotary Positional Embedding (RoPE) using separate 1D RoPE for height and width,
        and concatenates them.
        
        Extrapolation: If target_resolution is provided, a scaling factor 
            s = max(max(H_test, W_test) / L_train, 1.0) with L_train = sqrt(max_token_length)
        is computed. For methods such as NTK, VisionNTK, YaRN, or VisionYaRN,
        the inverse frequency vector is adjusted by dividing by s.
        
        Args:
            coords (torch.Tensor): Tensor [B, L, 2] with (h, w) coordinates.
            target_resolution (Optional[Tuple[int, int]]): (H_test, W_test) for extrapolation.
        Returns:
            torch.Tensor: Positional embeddings [B, L, hidden_dim].
        Raises:
            ValueError: If hidden_dim is not divisible by 4.
        """
        B, L, _ = coords.shape
        d_model: int = self.hidden_dim
        if d_model % 4 != 0:
            raise ValueError("hidden_dim must be divisible by 4 for 2D RoPE positional embedding.")
        d_each: int = d_model // 4  # Dimension for each sin-cos pair.
        device = coords.device
        # Create inverse frequency vector.
        i = torch.arange(0, d_each, device=device, dtype=torch.float32)
        inv_freq = 1.0 / (10000 ** (2 * i / d_each))
        # Split coordinates: shape [B, L, 1]
        pos_h = coords[..., 0:1].float()
        pos_w = coords[..., 1:2].float()
        # Adjust frequencies if target_resolution is provided.
        if target_resolution is not None:
            H_test, W_test = target_resolution
            L_train: float = math.sqrt(self.max_token_length)  # e.g., 16 for max_token_length=256
            s: float = max(max(H_test, W_test) / L_train, 1.0)
            if self.positional_interpolation_method in ["NTK", "VisionNTK", "YaRN", "VisionYaRN"]:
                inv_freq = inv_freq / s  # Adjust frequency (simple approximation for all these methods)
        # Compute angles.
        angle_h = pos_h * inv_freq  # [B, L, d_each]
        angle_w = pos_w * inv_freq
        # Compute sin and cos components.
        emb_h = torch.cat([torch.sin(angle_h), torch.cos(angle_h)], dim=-1)  # [B, L, 2*d_each]
        emb_w = torch.cat([torch.sin(angle_w), torch.cos(angle_w)], dim=-1)  # [B, L, 2*d_each]
        # Concatenate to form final positional embedding.
        pos_emb = torch.cat([emb_h, emb_w], dim=-1)  # [B, L, 4*d_each] == [B, L, d_model]
        return pos_emb

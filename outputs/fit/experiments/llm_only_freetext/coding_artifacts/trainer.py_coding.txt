"""
trainer.py

This module implements the Trainer class responsible for training the FiT diffusion model.
It performs the following:
  - Loads batches from a DataLoader (with images and token masks) generated by DatasetLoader.
  - For each batch, it encodes images using the pretrained VAE, patchifies and projects the latent codes,
    and then injects noise following a DDPM-based diffusion schedule.
  - It performs a forward pass through the transformer backbone, computes a masked mean squared error (MSE)
    loss between the predicted noise and the sampled noise (only over valid, non-padding tokens),
    and updates the model parameters via the AdamW optimizer.
  - An exponential moving average (EMA) of the model parameters is maintained.
  - Periodic logging and checkpointing are performed.
All configuration values are read from the provided config dictionary.
"""

import os
import math
import torch
import torch.nn as nn
import torch.optim as optim
import torch.nn.functional as F
from tqdm import tqdm
from typing import Any, Dict

class Trainer:
    """
    Trainer class responsible for the training loop of the FiT diffusion model.
    
    Attributes:
        model (nn.Module): The FiT model instance.
        dataloader (DataLoader): The PyTorch DataLoader from the DatasetLoader.
        config (Dict[str, Any]): Configuration dictionary (from config.yaml).
        device (torch.device): Computation device (GPU if available).
        optimizer (optim.Optimizer): Optimizer instance.
        ema_state (Dict[str, torch.Tensor]): EMA copy of model parameters.
        ema_decay (float): EMA decay coefficient.
        total_steps (int): Total number of training steps.
        checkpoint_interval (int): Interval in steps at which to save checkpoints.
        T (int): Total diffusion time steps (for noise schedule).
        alphas_bar (torch.Tensor): Pre-computed cumulative product of alphas for noise injection.
    """
    def __init__(self, model: nn.Module, dataloader: torch.utils.data.DataLoader, config: Dict[str, Any]) -> None:
        """
        Initialize the Trainer.

        Args:
            model (nn.Module): The FiT model instance.
            dataloader (torch.utils.data.DataLoader): Dataloader for training data.
            config (Dict[str, Any]): Configuration dictionary.
        """
        self.model: nn.Module = model
        self.dataloader: torch.utils.data.DataLoader = dataloader
        self.config: Dict[str, Any] = config

        # Determine the device: use CUDA if available.
        self.device: torch.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.model.to(self.device)

        # Set optimizer parameters from config (with default values)
        learning_rate: float = float(self.config.get("training", {}).get("learning_rate", 1e-4))
        weight_decay: float = float(self.config.get("training", {}).get("weight_decay", 0.0))
        self.optimizer: optim.Optimizer = optim.AdamW(self.model.parameters(), lr=learning_rate, weight_decay=weight_decay)

        # EMA settings: decay (default 0.9999) and initialize EMA state from model parameters.
        self.ema_decay: float = float(self.config.get("training", {}).get("ema_decay", 0.9999))
        self.ema_state: Dict[str, torch.Tensor] = {}
        for name, param in self.model.named_parameters():
            self.ema_state[name] = param.detach().clone().to(self.device)

        # Determine total training steps using model variant config.
        # For FiT-B/2, use training_steps_fit_b2 (default 400K); for FiT-XL/2, use training_steps_fit_xl2 (default 1.8M)
        model_variant: str = str(self.config.get("model", {}).get("variant", "B/2")).lower()
        if "xl" in model_variant:
            self.total_steps: int = int(self.config.get("training", {}).get("training_steps_fit_xl2", 1800000))
        else:
            self.total_steps: int = int(self.config.get("training", {}).get("training_steps_fit_b2", 400000))

        # Checkpoint interval (default 10000 steps)
        self.checkpoint_interval: int = int(self.config.get("training", {}).get("checkpoint_interval", 10000))

        # Setup diffusion noise schedule hyperparameters.
        # Default DDPM schedule parameters: T=1000, beta_start=0.0001, beta_end=0.02.
        self.T: int = 1000
        beta_start: float = 0.0001
        beta_end: float = 0.02
        betas: torch.Tensor = torch.linspace(beta_start, beta_end, self.T, dtype=torch.float32, device=self.device)
        alphas: torch.Tensor = 1.0 - betas
        # Compute cumulative products: alpha_bar[t] = prod_{i=0}^{t} alpha_i
        alphas_bar: torch.Tensor = torch.cumprod(alphas, dim=0)
        self.alphas_bar: torch.Tensor = alphas_bar  # shape: [T]

    def update_ema(self) -> None:
        """
        Update the exponential moving average (EMA) of the model parameters.
        For each parameter, EMA_param = ema_decay * EMA_param + (1 - ema_decay) * current_param.
        """
        for name, param in self.model.named_parameters():
            if param.requires_grad:
                self.ema_state[name] = self.ema_state[name] * self.ema_decay + param.detach() * (1.0 - self.ema_decay)

    def save_checkpoint(self, step: int) -> None:
        """
        Save a checkpoint of the current training state.

        Args:
            step (int): The current training step.
        """
        checkpoint = {
            "step": step,
            "model_state": self.model.state_dict(),
            "optimizer_state": self.optimizer.state_dict(),
            "ema_state": self.ema_state
        }
        checkpoint_filename: str = f"checkpoint_step_{step}.pt"
        torch.save(checkpoint, checkpoint_filename)
        print(f"Checkpoint saved: {checkpoint_filename}")

    def train(self) -> None:
        """
        Execute the training loop for the diffusion FiT model.
        """
        self.model.train()
        global_step: int = 0

        progress_bar = tqdm(total=self.total_steps, desc="Training Steps", ncols=80)
        
        # Create an iterator from dataloader that cycles indefinitely.
        dataloader_iter = iter(self.dataloader)
        
        while global_step < self.total_steps:
            try:
                batch = next(dataloader_iter)
            except StopIteration:
                dataloader_iter = iter(self.dataloader)
                batch = next(dataloader_iter)

            # Expect batch to have key "image" (preprocessed image tensor)
            images = batch.get("image", None)
            if images is None:
                raise ValueError("Batch does not contain 'image' key.")
            images = images.to(self.device)

            # --- Forward diffusion process --
            # 1. Encode images to latent representation using the VAE encoder.
            #    Note: We directly access self.model.vae_encoder and patchify_and_pad to compute tokens.
            latent = self.model.vae_encoder(images)  # latent shape: [B, C_latent, H_latent, W_latent]
            tokens, coords, validity_mask = self.model.patchify_and_pad(latent)
            # tokens shape: [B, T, token_dim]
            # validity_mask shape: [B, T] (1 for valid, 0 for padding)
            
            # 2. Project tokens to the transformer embedding dimension.
            x_clean = self.model.token_proj(tokens)  # [B, T, embed_dim]

            # 3. Compute 2D rotary positional embeddings from patch coordinates.
            pos_cos, pos_sin = self.model.compute_2d_rope(coords)  # Each: [B, T, embed_dim]

            # 4. Construct attention mask: valid tokens -> 0; padded tokens -> -1e4.
            attn_mask = (1.0 - validity_mask) * (-1e4)  # shape: [B, T]

            # 5. Diffusion noise injection:
            #    For each sample in the batch, sample a timestep t uniformly from [0, T-1]
            B = x_clean.size(0)
            t = torch.randint(low=0, high=self.T, size=(B,), device=self.device)
            # Retrieve cumulative product alpha_bar for each sample from pre-computed schedule
            sqrt_alpha_bar = self.alphas_bar[t].sqrt().view(B, 1, 1)  # shape: [B, 1, 1]
            sqrt_one_minus_alpha_bar = (1.0 - self.alphas_bar[t]).sqrt().view(B, 1, 1)  # shape: [B, 1, 1]
            
            # Sample noise from standard normal distribution (same shape as x_clean)
            noise = torch.randn_like(x_clean)
            # Compute noisy tokens:
            x_noisy = sqrt_alpha_bar * x_clean + sqrt_one_minus_alpha_bar * noise

            # --- Forward pass through the transformer backbone ---
            # Pass the noisy tokens (x_noisy) along with positional embeddings and attention mask
            predicted = x_noisy
            for block in self.model.transformer_blocks:
                predicted = block(predicted, pos_cos, pos_sin, attn_mask)
            predicted = self.model.final_norm(predicted)
            # The predicted output is expected to approximate the injected noise.

            # --- Loss computation (masked MSE) ---
            mse_loss = F.mse_loss(predicted, noise, reduction='none')  # shape: [B, T, embed_dim]
            # Expand validity_mask to match prediction dimensions.
            validity_mask_expanded = validity_mask.unsqueeze(-1)  # [B, T, 1]
            masked_loss = mse_loss * validity_mask_expanded
            # Sum loss and average over total number of valid elements.
            num_valid_tokens = validity_mask.sum() * predicted.size(-1)
            loss = masked_loss.sum() / (num_valid_tokens + 1e-8)

            # --- Backpropagation and optimizer update ---
            self.optimizer.zero_grad()
            loss.backward()
            self.optimizer.step()

            # --- EMA update ---
            self.update_ema()

            # --- Logging and checkpointing ---
            global_step += 1
            progress_bar.set_postfix({"Loss": f"{loss.item():.6f}", "Valid Tokens": f"{validity_mask.sum().item()}"})
            progress_bar.update(1)

            if global_step % self.checkpoint_interval == 0:
                self.save_checkpoint(global_step)

        progress_bar.close()
        print("Training finished.")

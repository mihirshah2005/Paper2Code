"""
dataset_loader.py

This module implements the dataset loader for graph datasets.
It loads a graph dataset (e.g., Pubmed), computes the adjacency matrix,
normalized Laplacian, and performs a truncated eigen-decomposition.
The computed components are packaged into a GraphData object.

Required packages:
    numpy==1.21.0
    torch==1.9.0
    scipy==1.7.0
    networkx==2.6.3
"""

import os
import numpy as np
import networkx as nx
import scipy.sparse as sp
from scipy.sparse.linalg import eigsh
from typing import Any, Optional


class GraphData:
    """
    A simple container for graph data required for synthetic graph distillation.
    
    Attributes:
        adjacency (np.ndarray): Adjacency matrix of the graph (shape: [N, N]).
        node_features (np.ndarray): Node feature matrix (shape: [N, d]).
        labels (np.ndarray): Node labels (shape: [N] or [N, 1]).
        laplacian (np.ndarray): Normalized graph Laplacian (shape: [N, N]).
        eigenvalues (np.ndarray): Concatenated eigenvalues from truncated eigen-decomposition (shape: [K_total, ]).
        eigenvectors (np.ndarray): Concatenated eigenvectors (shape: [N, K_total]).
        synthetic_size (int): The number of nodes in the distilled synthetic graph (N').
    """
    def __init__(
        self,
        adjacency: np.ndarray,
        node_features: np.ndarray,
        labels: np.ndarray,
        laplacian: np.ndarray,
        eigenvalues: np.ndarray,
        eigenvectors: np.ndarray,
        synthetic_size: int
    ) -> None:
        self.adjacency: np.ndarray = adjacency
        self.node_features: np.ndarray = node_features
        self.labels: np.ndarray = labels
        self.laplacian: np.ndarray = laplacian
        self.eigenvalues: np.ndarray = eigenvalues
        self.eigenvectors: np.ndarray = eigenvectors
        self.synthetic_size: int = synthetic_size


class DatasetLoader:
    """
    DatasetLoader loads a graph dataset based on the configuration.
    
    It:
      - Loads node features, labels, and build the graph from an edge list.
      - Computes the adjacency matrix and normalized Laplacian.
      - Performs truncated eigen-decomposition to extract a subset of eigenpairs.
      - Packages all components in a GraphData object.
    """
    def __init__(self, config: dict) -> None:
        self.config: dict = config
        dataset_config: dict = config.get("dataset", {})
        self.dataset_name: str = dataset_config.get("name", "Pubmed")
        self.split_method: str = dataset_config.get("split_method", "public")

        synth_config: dict = config.get("synth_graph", {})
        # compression_ratio is expected to be provided as a percentage value (e.g., 0.15 means 0.15%)
        self.compression_ratio: float = float(synth_config.get("compression_ratio", 0.15))
        self.r_k: float = float(synth_config.get("r_k", 0.9))

    def load_data(self) -> GraphData:
        """
        Loads and processes the graph dataset.

        Returns:
            GraphData: An object containing adjacency matrix, node features, labels,
                       normalized Laplacian, eigenvalues, eigenvectors and synthetic graph size.
        """
        # Currently, only "Pubmed" dataset is supported.
        if self.dataset_name.lower() == "pubmed":
            # Define the data directory and file path for Pubmed dataset
            data_dir: str = os.path.join(os.getcwd(), "data")
            pubmed_file: str = os.path.join(data_dir, "Pubmed.npz")
            if os.path.exists(pubmed_file):
                try:
                    # Expecting .npz file to contain 'features', 'labels', and 'adjacency'
                    data = np.load(pubmed_file, allow_pickle=True)
                    X: np.ndarray = data["features"]  # shape: [N, d]
                    Y: np.ndarray = data["labels"]      # shape: [N] or [N, 1]
                    # Load the adjacency matrix as a sparse matrix then convert it to CSR format
                    A_sparse: sp.csr_matrix = sp.csr_matrix(data["adjacency"])
                except Exception as e:
                    raise RuntimeError(f"Error loading Pubmed dataset from file: {e}")
            else:
                # If the dataset file is not found, load a synthetic dummy Pubmed dataset for testing.
                print("Pubmed dataset file not found. Generating a synthetic dummy Pubmed dataset for testing purposes.")
                # For testing, use a smaller graph
                N_dummy: int = 200  # number of nodes
                d_dummy: int = 50   # feature dimension
                # Generate a random graph using the Erdos-Renyi model
                G_dummy: nx.Graph = nx.erdos_renyi_graph(n=N_dummy, p=0.05, seed=42)
                # Ensure the graph is connected; if not, take the largest connected component.
                if not nx.is_connected(G_dummy):
                    connected_components = list(nx.connected_components(G_dummy))
                    largest_cc = max(connected_components, key=len)
                    G_dummy = G_dummy.subgraph(largest_cc).copy()
                X = np.random.rand(G_dummy.number_of_nodes(), d_dummy).astype(np.float32)
                # Assume three classes for testing
                Y = np.random.randint(0, 3, size=(G_dummy.number_of_nodes(),))
                A_sparse = nx.adjacency_matrix(G_dummy)
            
            # Convert the sparse adjacency matrix to a dense NumPy array of type float32
            A: np.ndarray = A_sparse.toarray().astype(np.float32)
        else:
            raise ValueError(f"Dataset {self.dataset_name} is not supported.")

        # Compute the degree for each node from the adjacency matrix A.
        degrees: np.ndarray = np.sum(A, axis=1)
        if np.any(degrees == 0):
            raise ValueError("One or more nodes have zero degree which violates the connected graph assumption.")

        # Compute D^(-0.5): the inverse square root of the degree.
        d_inv_sqrt: np.ndarray = np.power(degrees, -0.5)
        # Replace any infinite values with zero.
        d_inv_sqrt[np.isinf(d_inv_sqrt)] = 0.0
        
        # Form a diagonal matrix for D^(-0.5)
        D_inv_sqrt: np.ndarray = np.diag(d_inv_sqrt)
        # Identity matrix I
        N_total: int = A.shape[0]
        I: np.ndarray = np.eye(N_total, dtype=np.float32)
        # Compute the normalized Laplacian: L = I - D^(-0.5) * A * D^(-0.5)
        L: np.ndarray = I - D_inv_sqrt @ A @ D_inv_sqrt

        # Determine the synthetic graph size (N') based on the compression ratio.
        # The compression_ratio in config is given as a percentage (e.g., 0.15 means 0.15%).
        N_prime: int = max(1, int(N_total * (self.compression_ratio / 100.0)))

        # Set the number of eigenvectors to extract: total K = N_prime.
        # Compute K1 = int(N_prime * r_k) for the smallest eigenpairs (global structure)
        # and K2 = N_prime - K1 for the largest eigenpairs (local details).
        K1: int = int(N_prime * self.r_k)
        K2: int = N_prime - K1
        # Ensure that at least one eigenvector is selected for each group.
        if K1 <= 0:
            K1 = 1
        if K2 <= 0:
            K2 = 1
        K_total: int = K1 + K2

        # Convert the Laplacian to a sparse matrix format for efficient eigen-decomposition.
        L_sparse: sp.csr_matrix = sp.csr_matrix(L)
        # Compute the K1 smallest eigenpairs using eigsh.
        try:
            eigenvalues_small, eigenvectors_small = eigsh(L_sparse, k=K1, which='SM', tol=1e-3)
        except Exception as e:
            raise RuntimeError(f"Error during computing the smallest {K1} eigenpairs: {e}")
        
        # Compute the K2 largest eigenpairs using eigsh.
        try:
            eigenvalues_large, eigenvectors_large = eigsh(L_sparse, k=K2, which='LA', tol=1e-3)
        except Exception as e:
            raise RuntimeError(f"Error during computing the largest {K2} eigenpairs: {e}")
        
        # Concatenate the eigenvalues: smallest first, then largest.
        eigenvalues: np.ndarray = np.concatenate([eigenvalues_small, eigenvalues_large])
        # Concatenate the corresponding eigenvectors along the column axis.
        eigenvectors: np.ndarray = np.concatenate([eigenvectors_small, eigenvectors_large], axis=1)

        # Log key information for debugging.
        print(f"Loaded Graph Data:")
        print(f"  Number of nodes: {N_total}")
        print(f"  Adjacency matrix shape: {A.shape}")
        print(f"  Node feature matrix shape: {A.shape[0]} x {X.shape[1]}")
        print(f"  Laplacian matrix shape: {L.shape}")
        print(f"  Eigenvalues shape: {eigenvalues.shape}")
        print(f"  Eigenvectors shape: {eigenvectors.shape}")
        print(f"  Synthetic graph size (N'): {N_prime}")
        print(f"  Eigenpair split: K_total = {K_total} (K1 = {K1}, K2 = {K2})")

        # Package all computed components into a GraphData object.
        graph_data: GraphData = GraphData(
            adjacency=A,
            node_features=X,
            labels=Y,
            laplacian=L,
            eigenvalues=eigenvalues,
            eigenvectors=eigenvectors,
            synthetic_size=N_prime
        )
        return graph_data


# For standalone testing of this module, you can run the following code (it will only execute when this file is run directly).
if __name__ == "__main__":
    # Example configuration dictionary; in practice, this would be read from config.yaml.
    example_config: dict = {
        "training": {
            "epochs": 1500,
            "learning_rate_feat": 1e-05,
            "learning_rate_eigenvecs": 0.01,
            "tau1": 0,
            "tau2": 5,
            "optimizer": "adam",
            "loss_weights": {"alpha": 1.0, "beta": 1.0, "gamma": 1.0},
            "num_runs": 10
        },
        "synth_graph": {
            "compression_ratio": 0.15,  # 0.15% compression ratio
            "r_k": 0.9
        },
        "evaluation": {
            "gnn": {
                "spatial": {"n_layers": 2, "hidden_units": 256},
                "spectral": {"polynomial_order": 10, "hidden_units": 256}
            }
        },
        "dataset": {
            "name": "Pubmed",
            "split_method": "public"
        }
    }

    # Instantiate the DatasetLoader with the example configuration.
    loader = DatasetLoader(example_config)
    # Load the graph data.
    graph = loader.load_data()
    # Print out summary of the loaded GraphData.
    print("GraphData object created successfully.")
    print(f"Adjacency matrix type: {type(graph.adjacency)} with shape {graph.adjacency.shape}")
    print(f"Node features type: {type(graph.node_features)} with shape {graph.node_features.shape}")
    print(f"Labels type: {type(graph.labels)} with shape {graph.labels.shape}")
    print(f"Laplacian matrix type: {type(graph.laplacian)} with shape {graph.laplacian.shape}")
    print(f"Eigenvalues: {graph.eigenvalues}")
    print(f"Eigenvectors shape: {graph.eigenvectors.shape}")
    print(f"Synthetic graph size (N'): {graph.synthetic_size}")

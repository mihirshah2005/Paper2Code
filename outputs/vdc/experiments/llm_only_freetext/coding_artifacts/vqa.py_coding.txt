"""
vqa.py

This module implements the Visual Question Answering (VQA) class as part of the VDC system.
It loads and configures the multimodal model (Instruct-BLIP) using HuggingFace Transformers,
sets the model to evaluation mode, and provides the answer_question method that processes an
input image and question to return a human-readable answer.
"""

import time
import logging
from typing import Any

import torch
from PIL import Image
from transformers import BlipProcessor, BlipForConditionalGeneration

# Set up the logger for this module.
logger = logging.getLogger(__name__)
logger.setLevel(logging.INFO)
if not logger.handlers:
    # Add a console handler if not already present.
    console_handler = logging.StreamHandler()
    console_handler.setFormatter(logging.Formatter("%(asctime)s - %(levelname)s - %(message)s"))
    logger.addHandler(console_handler)


class VQA:
    """
    Visual Question Answering (VQA) module for the VDC system.

    Responsibilities:
      - Loads the Instruct-BLIP model and its associated processor/tokenizer based on the model configuration.
      - Sets the model to evaluation mode to ensure deterministic inference (e.g., no dropout).
      - Preprocesses the input image and text question, performs inference using the model,
        and decodes the output into a human-readable answer.
    
    Attributes:
        model_name (str): Name or identifier of the model to load.
        processor (BlipProcessor): Processor to preprocess inputs and decode outputs.
        model (BlipForConditionalGeneration): The loaded multimodal model.
        device (torch.device): The torch device ("cuda" if available, otherwise "cpu").
    """

    def __init__(self, model_config: dict) -> None:
        """
        Initialize the VQA instance with the model configuration.

        This method:
          - Retrieves the model name from the configuration (with default "Salesforce/instruct-blip").
          - Loads the processor and model using HuggingFace's from_pretrained API.
          - Sets the model to evaluation mode and moves it to the appropriate device.

        Args:
            model_config (dict): Configuration dictionary containing model settings.
                                 Expected key "model" with the model identifier.
        """
        # Set default model name if not provided.
        self.model_name: str = model_config.get("model", "Salesforce/instruct-blip")
        logger.info("Loading VQA model: %s", self.model_name)

        try:
            self.processor: BlipProcessor = BlipProcessor.from_pretrained(self.model_name)
            self.model: BlipForConditionalGeneration = BlipForConditionalGeneration.from_pretrained(self.model_name)
        except Exception as error:
            logger.error("Failed to load model or processor from '%s': %s", self.model_name, error)
            raise error

        # Determine the device to use (GPU if available).
        self.device: torch.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.model.to(self.device)

        # Set the model to evaluation mode to disable dropout and ensure determinism.
        self.model.eval()
        logger.info("Model loaded and set to evaluation mode on device: %s", self.device)

    def answer_question(self, image: Any, question: str) -> str:
        """
        Answer a given question based on the provided image using the VQA model.

        Steps:
          - Validates and converts the input image to a PIL Image if necessary.
          - Preprocesses the image and question using the BLIP processor.
          - Runs a forward pass in evaluation mode inside a torch.no_grad() context.
          - Decodes the generated tokens into a clean answer string.

        Args:
            image (Any): The input image. Can be a PIL Image or a numpy array representing the image.
            question (str): The question text to ask about the image.

        Returns:
            str: The answer generated by the model.
        """
        # Ensure the image is in PIL Image format.
        if not isinstance(image, Image.Image):
            try:
                image = Image.fromarray(image)
            except Exception as e:
                logger.error("Failed to convert the provided image to PIL format: %s", e)
                raise e

        # Prepare inputs using the processor.
        try:
            inputs = self.processor(images=image, text=question, return_tensors="pt")
            inputs = {k: v.to(self.device) for k, v in inputs.items()}
        except Exception as e:
            logger.error("Error during preprocessing of image/question pair: %s", e)
            raise e

        # Ensure model is in evaluation mode and run inference.
        with torch.no_grad():
            start_time: float = time.time()
            generated_ids = self.model.generate(**inputs)
            inference_time: float = time.time() - start_time
        logger.info("Inference completed in %.2f seconds for question: '%s'", inference_time, question)

        # Decode generated tokens to produce the final answer text.
        try:
            answer: str = self.processor.decode(generated_ids[0], skip_special_tokens=True)
        except Exception as e:
            logger.error("Error during decoding of the model output: %s", e)
            raise e

        final_answer: str = answer.strip()
        logger.info("Generated answer: '%s'", final_answer)
        return final_answer

"""vqa.py

This module implements the VQA (Visual Question Answering) class which wraps a multimodal
large language model (specifically Instruct-BLIP) to answer text questions given an input image.
The VQA class initializes the model and its associated processor, ensures that the input
image is preprocessed appropriately, and returns an answer text for downstream evaluation.

The expected workflow is:
  1. The VQA class is instantiated with model configuration details (from config.yaml).
  2. When answer_question(image, question) is called:
       - The image is checked (and converted if necessary) to a PIL.Image format.
       - The processor preprocesses the image and question.
       - The model generates an answer.
       - The answer is postprocessed and returned as a string.

Dependencies:
  - torch, torchvision, and PIL for image handling.
  - transformers for loading the Instruct-BLIP model (BlipProcessor and BlipForQuestionAnswering).
  - utils for configuration and logging.

Author: [Your Name]
Date: [Date]
"""

from typing import Any, Dict

import torch
import torchvision.transforms as transforms
from PIL import Image

from transformers import BlipProcessor, BlipForQuestionAnswering

from utils import load_config, get_logger


class VQA:
    """
    VQA (Visual Question Answering) class to obtain an answer text for a given image-question pair.
    
    Attributes:
        processor (BlipProcessor): Preprocessing processor for the Instruct-BLIP model.
        model (BlipForQuestionAnswering): The Instruct-BLIP model for visual question answering.
        device (torch.device): The device (CPU or CUDA) on which the model is running.
        image_transform (transforms.Compose): A torchvision transform pipeline to preprocess images,
            if needed.
        logger: Logger instance for debugging and info messages.
    """

    def __init__(self, model_config: Dict[str, Any]) -> None:
        """
        Initializes the VQA instance using the provided model configuration.
        Loads the Instruct-BLIP model and its processor, and sets up the image preprocessing pipeline.

        Args:
            model_config (Dict[str, Any]): Dictionary with model configuration parameters.
                Expected to include at least the key "model" (e.g., "Instruct-BLIP").
                If not provided, defaults to {"model": "Instruct-BLIP"}.
        """
        # Ensure we have a model configuration dictionary.
        if not model_config:
            model_config = {"model": "Instruct-BLIP"}

        # Initialize logger.
        self.logger = get_logger("VQA")

        # Retrieve the model name from configuration; default to "Instruct-BLIP" if not specified.
        self.model_name: str = model_config.get("model", "Instruct-BLIP")
        self.logger.info(f"Loading VQA model: {self.model_name}")

        # Determine device: use GPU if available, else CPU.
        self.device: torch.device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
        self.logger.info(f"Using device: {self.device}")

        # Load the processor and model from the transformers library.
        try:
            self.processor: BlipProcessor = BlipProcessor.from_pretrained(self.model_name)
            self.model: BlipForQuestionAnswering = BlipForQuestionAnswering.from_pretrained(self.model_name)
        except Exception as e:
            self.logger.error(f"Error loading the Instruct-BLIP model: {e}")
            raise e

        # Move the model to the specified device and set to evaluation mode.
        self.model.to(self.device)
        self.model.eval()

        # Define an image transformation pipeline (fallback if needed).
        # This pipeline resizes the image to 224x224 and normalizes using ImageNet statistics.
        self.image_transform = transforms.Compose([
            transforms.Resize((224, 224)),
            transforms.ToTensor(),
            transforms.Normalize(mean=[0.485, 0.456, 0.406],
                                 std=[0.229, 0.224, 0.225])
        ])

    def answer_question(self, image: Any, question: str) -> str:
        """
        Answers the given text question based on the provided image using the Instruct-BLIP model.
        The image is preprocessed as necessary and then both image and question are passed to the model.

        Args:
            image (Any): Input image which should be either a PIL.Image or a torch.Tensor.
            question (str): The text question to be answered.

        Returns:
            str: The answer text generated by the model. In case of an error, returns an empty string.
        """
        try:
            # If the input image is not a PIL Image, attempt to convert if it is a torch.Tensor.
            if not isinstance(image, Image.Image):
                if isinstance(image, torch.Tensor):
                    image = transforms.ToPILImage()(image)
                else:
                    error_msg = "Unsupported image format. Expected PIL.Image or torch.Tensor."
                    self.logger.error(error_msg)
                    raise ValueError(error_msg)

            # Use the BLIP processor to prepare inputs.
            inputs = self.processor(image, question, return_tensors="pt")
            # Move all tensor inputs to the designated device.
            inputs = {key: value.to(self.device) for key, value in inputs.items()}

            # Generate answer using the model.
            with torch.no_grad():
                output_ids = self.model.generate(**inputs)
            # Decode the generated ids to text.
            answer: str = self.processor.decode(output_ids[0], skip_special_tokens=True).strip()
            self.logger.info(f"Question: {question} | Answer: {answer}")
            return answer

        except Exception as e:
            self.logger.error(f"Error in answer_question: {e}")
            return ""

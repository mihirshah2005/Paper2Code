"""
model.py

This module implements the ModelManager class along with the necessary backbone models for
computer vision (CV), natural language processing (NLP), and graph machine learning domains. 
The ModelManager provides a unified interface to build a model instance based on the domain, 
model name, number of classes, and additional dataset-specific parameters.

The supported models are:

• CV Models:
    - LeNet: A simple convolutional network.
    - ResNet-18: Based on torchvision's implementation with a modified final fully connected layer.
    - ViT: A minimal Vision Transformer implementation with patch embedding and transformer encoder.

• NLP Models:
    - LSTM: A custom LSTM-based classifier with an embedding layer.
    - BERT: Built on Hugging Face's BertForSequenceClassification.
    - GPT2: Built on Hugging Face's GPT2ForSequenceClassification, with pad token configuration.

• Graph Models:
    - GCN: A Graph Convolutional Network using two GCNConv layers and global pooling.
    - GAT: A Graph Attention Network using two GATConv layers and global pooling.
    - GIN: A Graph Isomorphism Network using multiple GINConv layers and global pooling.

All models are designed to output logits for classification based on the number of target classes.
Configuration parameters are passed via a configuration dictionary (e.g., loaded with Config from config.py).
"""

import torch
import torch.nn as nn
import torch.nn.functional as F
import torchvision.models as tv_models

from transformers import BertForSequenceClassification, GPT2ForSequenceClassification

# Imports for graph models from torch_geometric
from torch_geometric.nn import GCNConv, GATConv, GINConv, global_mean_pool


# --------------------- CV Models --------------------- #
class LeNet(nn.Module):
    """A simple implementation of LeNet for image classification."""

    def __init__(self, num_classes: int, input_channels: int = 3) -> None:
        """
        Initializes LeNet architecture.

        Args:
            num_classes (int): Number of output classes.
            input_channels (int, optional): Number of input channels. Defaults to 3.
        """
        super(LeNet, self).__init__()
        self.conv1 = nn.Conv2d(in_channels=input_channels, out_channels=6, kernel_size=5)
        self.pool = nn.MaxPool2d(kernel_size=2, stride=2)
        self.conv2 = nn.Conv2d(in_channels=6, out_channels=16, kernel_size=5)
        # Assuming input image size 32x32:
        # conv1 output: (32-5+1)=28, then pool -> 14
        # conv2 output: (14-5+1)=10, then pool -> 5
        self.fc1 = nn.Linear(16 * 5 * 5, 120)
        self.fc2 = nn.Linear(120, 84)
        self.fc3 = nn.Linear(84, num_classes)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Forward pass for LeNet.

        Args:
            x (torch.Tensor): Input tensor of shape (B, C, H, W).

        Returns:
            torch.Tensor: Logits tensor of shape (B, num_classes).
        """
        x = F.relu(self.conv1(x))
        x = self.pool(x)
        x = F.relu(self.conv2(x))
        x = self.pool(x)
        x = x.view(x.size(0), -1)  # Flatten
        x = F.relu(self.fc1(x))
        x = F.relu(self.fc2(x))
        logits = self.fc3(x)
        return logits


class ViT(nn.Module):
    """
    A minimal Vision Transformer (ViT) implementation for image classification.
    This implementation divides the image into patches via a convolutional projection,
    applies a transformer encoder, and uses a classification token for final prediction.
    """

    def __init__(
        self,
        image_size: int = 32,
        patch_size: int = 4,
        in_channels: int = 3,
        num_classes: int = 10,
        embed_dim: int = 128,
        num_layers: int = 6,
        num_heads: int = 8,
        mlp_dim: int = 256,
        dropout: float = 0.1
    ) -> None:
        """
        Initializes the ViT model.

        Args:
            image_size (int, optional): Height/width of the input image. Defaults to 32.
            patch_size (int, optional): Size of each patch. Defaults to 4.
            in_channels (int, optional): Number of input channels. Defaults to 3.
            num_classes (int, optional): Number of output classes. Defaults to 10.
            embed_dim (int, optional): Dimension of the patch embeddings. Defaults to 128.
            num_layers (int, optional): Number of transformer encoder layers. Defaults to 6.
            num_heads (int, optional): Number of attention heads. Defaults to 8.
            mlp_dim (int, optional): Dimension of the feed-forward network in the encoder. Defaults to 256.
            dropout (float, optional): Dropout rate. Defaults to 0.1.
        """
        super(ViT, self).__init__()
        self.patch_size: int = patch_size
        self.num_patches: int = (image_size // patch_size) ** 2
        # Patch embedding using a convolution with kernel size equal to patch size and stride equal to patch size.
        self.proj = nn.Conv2d(in_channels, embed_dim, kernel_size=patch_size, stride=patch_size)
        # Classification token
        self.cls_token = nn.Parameter(torch.zeros(1, 1, embed_dim))
        # Positional embedding for all tokens (patch tokens + classification token)
        self.pos_embed = nn.Parameter(torch.zeros(1, self.num_patches + 1, embed_dim))
        # Transformer encoder layers
        encoder_layer = nn.TransformerEncoderLayer(d_model=embed_dim, nhead=num_heads, dim_feedforward=mlp_dim, dropout=dropout)
        self.transformer = nn.TransformerEncoder(encoder_layer, num_layers=num_layers)
        self.dropout = nn.Dropout(dropout)
        self.fc = nn.Linear(embed_dim, num_classes)
        self._init_weights()

    def _init_weights(self) -> None:
        """Initializes weights for positional embeddings, classification token, and final layer."""
        nn.init.trunc_normal_(self.pos_embed, std=0.02)
        nn.init.trunc_normal_(self.cls_token, std=0.02)
        nn.init.xavier_uniform_(self.fc.weight)
        if self.fc.bias is not None:
            nn.init.constant_(self.fc.bias, 0)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Forward pass for ViT.

        Args:
            x (torch.Tensor): Input tensor of shape (B, C, H, W).

        Returns:
            torch.Tensor: Logits tensor of shape (B, num_classes).
        """
        B = x.shape[0]
        x = self.proj(x)  # Shape: (B, embed_dim, H/patch, W/patch)
        x = x.flatten(2)  # Shape: (B, embed_dim, num_patches)
        x = x.transpose(1, 2)  # Shape: (B, num_patches, embed_dim)
        # Expand and prepend the classification token.
        cls_tokens = self.cls_token.expand(B, -1, -1)
        x = torch.cat((cls_tokens, x), dim=1)  # Shape: (B, num_patches + 1, embed_dim)
        x = x + self.pos_embed
        x = self.dropout(x)
        x = self.transformer(x)  # Shape: (B, num_patches + 1, embed_dim)
        cls_output = x[:, 0]  # Use the classification token
        logits = self.fc(cls_output)
        return logits


# --------------------- NLP Models --------------------- #
class LSTMClassifier(nn.Module):
    """
    A simple LSTM-based classifier for NLP tasks.
    It includes an embedding layer, an LSTM encoder, and a fully connected classification layer.
    """

    def __init__(
        self,
        num_classes: int,
        vocab_size: int = 10000,
        embedding_dim: int = 300,
        hidden_dim: int = 128,
        num_layers: int = 1,
        bidirectional: bool = False,
        dropout: float = 0.5
    ) -> None:
        """
        Initializes the LSTMClassifier.

        Args:
            num_classes (int): Number of output classes.
            vocab_size (int, optional): Vocabulary size. Defaults to 10000.
            embedding_dim (int, optional): Dimension of the embeddings. Defaults to 300.
            hidden_dim (int, optional): Dimension of the LSTM hidden state. Defaults to 128.
            num_layers (int, optional): Number of LSTM layers. Defaults to 1.
            bidirectional (bool, optional): If True, use a bidirectional LSTM. Defaults to False.
            dropout (float, optional): Dropout rate for the classifier. Defaults to 0.5.
        """
        super(LSTMClassifier, self).__init__()
        self.embedding = nn.Embedding(vocab_size, embedding_dim)
        self.lstm = nn.LSTM(
            input_size=embedding_dim,
            hidden_size=hidden_dim,
            num_layers=num_layers,
            batch_first=True,
            bidirectional=bidirectional
        )
        self.dropout = nn.Dropout(dropout)
        fc_input_dim = hidden_dim * 2 if bidirectional else hidden_dim
        self.fc = nn.Linear(fc_input_dim, num_classes)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        """
        Forward pass for the LSTM classifier.

        Args:
            x (torch.Tensor): Input tensor of token indices, shape (B, seq_length).

        Returns:
            torch.Tensor: Logits tensor of shape (B, num_classes).
        """
        embedded = self.embedding(x)  # Shape: (B, seq_length, embedding_dim)
        lstm_out, (hn, _) = self.lstm(embedded)
        if self.lstm.bidirectional:
            final_hidden = torch.cat((hn[-2], hn[-1]), dim=1)
        else:
            final_hidden = hn[-1]
        logits = self.fc(self.dropout(final_hidden))
        return logits


# --------------------- Graph Models --------------------- #
class GCNModel(nn.Module):
    """
    A Graph Convolutional Network (GCN) model for graph classification.
    Uses two GCNConv layers followed by global mean pooling and a linear classification head.
    """

    def __init__(self, num_features: int, hidden_dim: int = 64, num_classes: int = 2) -> None:
        """
        Initializes the GCN model.

        Args:
            num_features (int): Number of node features.
            hidden_dim (int, optional): Hidden dimension for the GCN layers. Defaults to 64.
            num_classes (int, optional): Number of output classes. Defaults to 2.
        """
        super(GCNModel, self).__init__()
        self.conv1 = GCNConv(num_features, hidden_dim)
        self.conv2 = GCNConv(hidden_dim, hidden_dim)
        self.fc = nn.Linear(hidden_dim, num_classes)

    def forward(self, data: torch.Tensor) -> torch.Tensor:
        """
        Forward pass for the GCN model.

        Args:
            data (torch_geometric.data.Data): Graph data object containing x, edge_index, and batch.

        Returns:
            torch.Tensor: Logits tensor of shape (B, num_classes).
        """
        x, edge_index, batch = data.x, data.edge_index, data.batch
        x = self.conv1(x, edge_index)
        x = F.relu(x)
        x = self.conv2(x, edge_index)
        x = F.relu(x)
        x = global_mean_pool(x, batch)
        logits = self.fc(x)
        return logits


class GATModel(nn.Module):
    """
    A Graph Attention Network (GAT) model for graph classification.
    Uses two GATConv layers with an attention mechanism followed by global mean pooling and a linear head.
    """

    def __init__(self, num_features: int, hidden_dim: int = 64, num_classes: int = 2, heads: int = 8) -> None:
        """
        Initializes the GAT model.

        Args:
            num_features (int): Number of node features.
            hidden_dim (int, optional): Hidden dimension for the GAT layers. Defaults to 64.
            num_classes (int, optional): Number of output classes. Defaults to 2.
            heads (int, optional): Number of attention heads in the first layer. Defaults to 8.
        """
        super(GATModel, self).__init__()
        self.conv1 = GATConv(num_features, hidden_dim, heads=heads, dropout=0.6)
        # For second layer, output dimension is hidden_dim (using 1 head with concat=False)
        self.conv2 = GATConv(hidden_dim * heads, hidden_dim, heads=1, concat=False, dropout=0.6)
        self.fc = nn.Linear(hidden_dim, num_classes)

    def forward(self, data: torch.Tensor) -> torch.Tensor:
        """
        Forward pass for the GAT model.

        Args:
            data (torch_geometric.data.Data): Graph data object with x, edge_index, and batch.

        Returns:
            torch.Tensor: Logits tensor of shape (B, num_classes).
        """
        x, edge_index, batch = data.x, data.edge_index, data.batch
        x = self.conv1(x, edge_index)
        x = F.relu(x)
        x = self.conv2(x, edge_index)
        x = F.relu(x)
        x = global_mean_pool(x, batch)
        logits = self.fc(x)
        return logits


class GINModel(nn.Module):
    """
    A Graph Isomorphism Network (GIN) model for graph classification.
    Uses multiple GINConv layers followed by global mean pooling and a final linear classifier.
    """

    def __init__(self, num_features: int, hidden_dim: int = 64, num_classes: int = 2, num_layers: int = 3) -> None:
        """
        Initializes the GIN model.

        Args:
            num_features (int): Number of node features.
            hidden_dim (int, optional): Hidden dimension for the MLPs in GINConv. Defaults to 64.
            num_classes (int, optional): Number of output classes. Defaults to 2.
            num_layers (int, optional): Number of GINConv layers. Defaults to 3.
        """
        super(GINModel, self).__init__()
        # Build a series of GINConv layers with an MLP as update function.
        nn1 = nn.Sequential(nn.Linear(num_features, hidden_dim), nn.ReLU(), nn.Linear(hidden_dim, hidden_dim))
        self.conv1 = GINConv(nn1)
        nn2 = nn.Sequential(nn.Linear(hidden_dim, hidden_dim), nn.ReLU(), nn.Linear(hidden_dim, hidden_dim))
        self.conv2 = GINConv(nn2)
        nn3 = nn.Sequential(nn.Linear(hidden_dim, hidden_dim), nn.ReLU(), nn.Linear(hidden_dim, hidden_dim))
        self.conv3 = GINConv(nn3)
        self.fc = nn.Linear(hidden_dim, num_classes)
        self.num_layers = num_layers

    def forward(self, data: torch.Tensor) -> torch.Tensor:
        """
        Forward pass for the GIN model.

        Args:
            data (torch_geometric.data.Data): Graph data object with x, edge_index, and batch.

        Returns:
            torch.Tensor: Logits tensor of shape (B, num_classes).
        """
        x, edge_index, batch = data.x, data.edge_index, data.batch
        x = self.conv1(x, edge_index)
        x = F.relu(x)
        x = self.conv2(x, edge_index)
        x = F.relu(x)
        x = self.conv3(x, edge_index)
        x = F.relu(x)
        x = global_mean_pool(x, batch)
        logits = self.fc(x)
        return logits


# --------------------- ModelManager --------------------- #
class ModelManager:
    """
    ModelManager instantiates backbone models based on the experiment configuration.
    It supports CV, NLP, and Graph domains and adjusts model input/output layers based on dataset properties.
    """

    def __init__(self, config: dict) -> None:
        """
        Initializes the ModelManager with a configuration dictionary.

        Args:
            config (dict): Experiment configuration loaded from config.yaml.
        """
        self.config = config

    def build_model(self, domain: str, model_name: str, num_classes: int, **kwargs) -> nn.Module:
        """
        Builds and returns a model instance based on the domain, model name, and dataset properties.

        Args:
            domain (str): Domain identifier ("cv", "nlp", "graph").
            model_name (str): Model identifier (e.g., "LeNet", "ResNet-18", "ViT" for CV;
                              "LSTM", "BERT", "GPT2" for NLP; "GCN", "GAT", "GIN" for Graph).
            num_classes (int): Number of output classes.
            **kwargs: Additional domain-specific parameters (e.g., input_channels, image_size, num_features).

        Returns:
            nn.Module: An instantiated PyTorch model.
        
        Raises:
            ValueError: If the domain or model_name is not supported.
        """
        domain_lower: str = domain.lower()
        model_name_lower: str = model_name.lower()
        if domain_lower == "cv":
            return self._build_cv_model(model_name_lower, num_classes, **kwargs)
        elif domain_lower == "nlp":
            return self._build_nlp_model(model_name_lower, num_classes, **kwargs)
        elif domain_lower == "graph":
            return self._build_graph_model(model_name_lower, num_classes, **kwargs)
        else:
            raise ValueError(f"Unsupported domain: {domain}")

    def _build_cv_model(self, model_name: str, num_classes: int, **kwargs) -> nn.Module:
        """
        Builds a computer vision model instance.

        Args:
            model_name (str): CV model identifier ("lenet", "resnet-18", "vit").
            num_classes (int): Number of output classes.
            **kwargs: Additional parameters such as input_channels, image_size.

        Returns:
            nn.Module: An instantiated CV model.
        """
        if model_name == "lenet":
            input_channels: int = kwargs.get("input_channels", 3)
            return LeNet(num_classes=num_classes, input_channels=input_channels)
        elif model_name == "resnet-18":
            # Use torchvision's resnet18; default pretrained is set to False.
            model = tv_models.resnet18(pretrained=False)
            in_features: int = model.fc.in_features
            model.fc = nn.Linear(in_features, num_classes)
            return model
        elif model_name == "vit":
            image_size: int = kwargs.get("image_size", 32)
            input_channels: int = kwargs.get("input_channels", 3)
            # Default hyperparameters for minimal ViT; can be adjusted via kwargs if needed.
            embed_dim: int = kwargs.get("embed_dim", 128)
            num_layers: int = kwargs.get("num_layers", 6)
            num_heads: int = kwargs.get("num_heads", 8)
            mlp_dim: int = kwargs.get("mlp_dim", 256)
            dropout: float = kwargs.get("dropout", 0.1)
            return ViT(
                image_size=image_size,
                patch_size=4,
                in_channels=input_channels,
                num_classes=num_classes,
                embed_dim=embed_dim,
                num_layers=num_layers,
                num_heads=num_heads,
                mlp_dim=mlp_dim,
                dropout=dropout
            )
        else:
            raise ValueError(f"Unsupported CV model: {model_name}")

    def _build_nlp_model(self, model_name: str, num_classes: int, **kwargs) -> nn.Module:
        """
        Builds a natural language processing model instance.

        Args:
            model_name (str): NLP model identifier ("lstm", "bert", "gpt2").
            num_classes (int): Number of output classes.
            **kwargs: Additional parameters such as vocab_size, embedding_dim, hidden_dim.

        Returns:
            nn.Module: An instantiated NLP model.
        """
        if model_name == "lstm":
            vocab_size: int = kwargs.get("vocab_size", 10000)
            embedding_dim: int = kwargs.get("embedding_dim", 300)
            hidden_dim: int = kwargs.get("hidden_dim", 128)
            num_layers: int = kwargs.get("num_layers", 1)
            bidirectional: bool = kwargs.get("bidirectional", False)
            dropout: float = kwargs.get("dropout", 0.5)
            return LSTMClassifier(
                num_classes=num_classes,
                vocab_size=vocab_size,
                embedding_dim=embedding_dim,
                hidden_dim=hidden_dim,
                num_layers=num_layers,
                bidirectional=bidirectional,
                dropout=dropout
            )
        elif model_name == "bert":
            return BertForSequenceClassification.from_pretrained("bert-base-uncased", num_labels=num_classes)
        elif model_name == "gpt2":
            model = GPT2ForSequenceClassification.from_pretrained("gpt2", num_labels=num_classes)
            if not hasattr(model.config, "pad_token_id") or model.config.pad_token_id is None:
                model.config.pad_token_id = 0
            return model
        else:
            raise ValueError(f"Unsupported NLP model: {model_name}")

    def _build_graph_model(self, model_name: str, num_classes: int, **kwargs) -> nn.Module:
        """
        Builds a graph model instance.

        Args:
            model_name (str): Graph model identifier ("gcn", "gat", "gin").
            num_classes (int): Number of output classes.
            **kwargs: Additional parameters such as num_features, hidden_dim, heads, num_layers.

        Returns:
            nn.Module: An instantiated graph model.
        """
        if model_name == "gcn":
            num_features: int = kwargs.get("num_features", 16)
            hidden_dim: int = kwargs.get("hidden_dim", 64)
            return GCNModel(num_features=num_features, hidden_dim=hidden_dim, num_classes=num_classes)
        elif model_name == "gat":
            num_features: int = kwargs.get("num_features", 16)
            hidden_dim: int = kwargs.get("hidden_dim", 64)
            heads: int = kwargs.get("heads", 8)
            return GATModel(num_features=num_features, hidden_dim=hidden_dim, num_classes=num_classes, heads=heads)
        elif model_name == "gin":
            num_features: int = kwargs.get("num_features", 16)
            hidden_dim: int = kwargs.get("hidden_dim", 64)
            num_layers: int = kwargs.get("num_layers", 3)
            return GINModel(num_features=num_features, hidden_dim=hidden_dim, num_classes=num_classes, num_layers=num_layers)
        else:
            raise ValueError(f"Unsupported Graph model: {model_name}")
